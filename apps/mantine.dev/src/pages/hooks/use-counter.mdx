## Definition

```tsx
export interface UseCounterOptions {
  min?: number;
  max?: number;
}

export interface UseCounterReturnValue {
  increment: () => void;
  decrement: () => void;
  set: (value: number) => void;
  reset: () => void;
}

function useCounter(
  initialValue?: number,
  options?: UseCounterOptions,
): [number, UseCounterReturnValue]
```

## Exported types

`UseCounterOptions` and `UseCounterReturnValue` types are exported from `@mantine/hooks` package,
you can import them in your application:

```tsx
import type { UseCounterOptions, UseCounterReturnValue } from '@mantine/hooks';
```
